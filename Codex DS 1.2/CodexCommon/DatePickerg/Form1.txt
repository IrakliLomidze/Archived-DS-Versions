		private int  StringToMonth(String Month)
		{
			int Ret = 0;;
			switch (Month) 
			{
				case "июмеюри"   : Ret = 1; break;
				case "хдадреюки" : Ret = 2; break;
				case "люрти"     : Ret = 3; break;
				case "юприки"    : Ret = 4; break;
				case "люиси"     : Ret = 5; break;
				case "иемиси"    : Ret = 6; break;
				case "иекиси"    : Ret = 7; break;
				case "юбеисто"   : Ret = 8; break;
				case "сдьтдладри": Ret = 9; break;
				case "оьтоладри" : Ret = 10; break;
				case "модладри"  : Ret = 11; break;
				case "цдйдладри" : Ret = 12; break;
			}
			return Ret;

		}
